177    /**
178     * Parse command line arguments into a Config object
179     *
180     * @param args command line arguments
181     * @return the parsed config file or command line args.
182     */
183    protected Config parseConfiguration(final String[] args) {
184        // first see if they've specified a config file
185        CommandLine c = null;
186        Config config = null;
187        try {
188            c = parseConfigArgs(configFileOptions, args);
189            config = parseConfigFileOptions(c);
190            addSharedOptions(c, config);
191        } catch (final ParseException ignore) {
192            logger.debug("Command line argments weren't valid for specifying a config file.");
193        }
194        if (config == null) {
195            // check for presence of the help flag
196            if (helpFlagged(args)) {
197                printHelp(null);
198            }
199
200            try {
201                c = parseConfigArgs(configOptions, args);
202                config = parseConfigurationArgs(c);
203                addSharedOptions(c, config);
204            } catch (final ParseException e) {
205                printHelp("Error parsing args: " + e.getMessage());
206            }
207        }
208
209        validateConfig(config);
210
211        // Write command line options to disk
212        saveConfig(config);
213
214        return config;
215    }
216
217    /**
218     * Validation routines for checking interdependent configuration parameters.
219     * @param config
220     */
221    private void validateConfig(final Config config) {
222        // verify bagit config
223        if (config.getBagProfile() != null &&
224                config.getBagConfigPath() == null) {
225            throw new RuntimeException("A bagit config path must be set when using a bagit profile.");
226        }
227
228        if (config.getBagProfile() == null &&
229                config.getBagConfigPath() != null) {
230            throw new RuntimeException("A bagit profile must be set when you set a bagit config.");
231        }
232
233    }

235    /**
236     * Check for a help flag
237     *
238     * @param args command line arguments
239     * @return whether the help flag was found.
240     */
241    private boolean helpFlagged(final String[] args) {
242        return Stream.of(args).anyMatch(x -> x.equals("-h") || x.equals("--help"));
243    }

245    /**
246     * Parse command line options based on the provide Options
247     *
248     * @param configOptions valid set of Options
249     * @param args command line arguments
250     * @return the list of option and values
251     * @throws ParseException if invalid/missing option is found
252     */
253    private static CommandLine parseConfigArgs(final Options configOptions, final String[] args) throws ParseException {
254        return new DefaultParser().parse(configOptions, args);
255    }

257    /**
258     * This method tries to parse the configuration file, if that option was provided
259     *
260     * @param args from command line
261     * @return Config or null if no config file option was provided
262     */
263    private Config parseConfigFileOptions(final CommandLine cmd) {
264        final Config config = retrieveConfig(new File(cmd.getOptionValue('c')));
265        addSharedOptions(cmd, config);
266        return config;
267    }

269    /**
270     * This method parses the provided configFile into its equivalent command-line args
271     *
272     * @param configFile containing config args
273     * @return Array of args
274     */
275    private Config retrieveConfig(final File configFile) {
276        if (!configFile.exists()) {
277            printHelp("Configuration file does not exist: " + configFile);
278        }
279        logger.debug("Loading configuration file: {}", configFile);
280        try {
281            final YamlReader reader = new YamlReader(new FileReader(configFile));
282            @SuppressWarnings("unchecked")
283            final Map<String, String> configVars = (HashMap<String, String>) reader.read();
284            return configFromFile(configVars);
285
286        } catch (final IOException | java.text.ParseException e) {
287            throw new RuntimeException("Unable to read configuration file due to: " + e.getMessage(), e);
288        }
289    }
290
291    /**
292     * This method parses the command-line args
293     *
294     * @param cmd command line options
295     * @return Config
296     */
297    private Config parseConfigurationArgs(final CommandLine cmd) {
298        final Config config = new Config();
299
300        // Inspect Mode option
301        final String mode = cmd.getOptionValue('m');
302        if (!mode.equalsIgnoreCase("import") && !mode.equalsIgnoreCase("export")) {
303            printHelp("Invalid 'mode' option: " + mode);
304        }
305
306        config.setMode(mode);
307        config.setResource(cmd.getOptionValue('r'));
308        config.setBaseDirectory(cmd.getOptionValue('d'));
309        config.setIncludeBinaries(cmd.hasOption('b'));
310
311        final String rdfLanguage = cmd.getOptionValue('l');
312        if (rdfLanguage != null) {
313            config.setRdfLanguage(rdfLanguage);
314        }
315        config.setSource(cmd.getOptionValue('s'));
316        if (cmd.getOptionValues('p') != null) {
317            config.setPredicates(cmd.getOptionValues('p'));
318        }
319
320        config.setBagProfile(cmd.getOptionValue('g'));
321        config.setBagConfigPath(cmd.getOptionValue('G'));
322
323        config.setAuditLog(cmd.hasOption('a'));
324
325        return config;
326    }

330    /**
331     * This method add/updates the values of any options that may be
332     * valid in either scenario (config file or fully command line)
333     *
334     * @param cmd a parsed command line
335     * @return Config the config which may be updated
336     */
337    private void addSharedOptions(final CommandLine cmd, final Config config) {
338        final String user = cmd.getOptionValue("user");
339        if (user != null) {
340            if (user.indexOf(':') == -1) {
341                printHelp("user option must be in the format username:password");
342            } else {
343                config.setUsername(user.substring(0, user.indexOf(':')));
344                config.setPassword(user.substring(user.indexOf(':') + 1));
345            }
346        }
347    }
348
349    /**
350     * This method writes the configuration file to disk.  The current
351     * implementation omits the user/password information.
352     * @param args to be persisted
353     */
354    private void saveConfig(final Config config) {
355        final File configFile = new File(System.getProperty("java.io.tmpdir"), CONFIG_FILE_NAME);
356
357        // Leave existing config file alone
358        if (configFile.exists()) {
359            logger.info("Configuration file exists, new file will NOT be created: {}", configFile.getPath());
360            return;
361        }
362
363        // Write config to file
364        try {
365            final YamlWriter writer = new YamlWriter(new FileWriter(configFile));
366            logger.debug("YAML output is ({})", getMap(config).toString());
367            writer.write(getMap(config));
368            writer.close();
369            logger.info("Saved configuration to: {}", configFile.getPath());
370
371        } catch (final IOException e) {
372            throw new RuntimeException("Unable to write configuration file due to: " + e.getMessage(), e);
373        }
374    }

376    /**
377     * Parse command-line arguments.
378     * @param args Command-line arguments
379     * @return A configured Importer or Exporter instance.
380    **/
381    public TransferProcess parse(final String[] args) {
382        final Config config = parseConfiguration(args);
383        if (config.isImport()) {
384            return new Importer(config, clientBuilder());
385        } else if (config.isExport()) {
386            return new Exporter(config, clientBuilder());
387        }
388        throw new IllegalArgumentException("Invalid mode parameter");
389    }

400    /**
401     * Print help/usage information
402     *
403     * @param message the message or null for none
404     */
405    private void printHelp(final String message) {
406        final HelpFormatter formatter = new HelpFormatter();
407        final PrintWriter writer = new PrintWriter(System.out);
408        if (message != null) {
409            writer.println("\n-----------------------\n" + message + "\n-----------------------\n");
410        }
411
412        writer.println("Running Import/Export Utility from command line arguments");
413        formatter.printHelp(writer, 80, "java -jar import-export-driver.jar", "", configOptions, 4, 4, "", true);
414
415        writer.println("\n--- or ---\n");
416
417        writer.println("Running Import/Export Utility from configuration file");
418        formatter.printHelp(writer, 80, "java -jar import-export-driver.jar", "", configFileOptions, 4, 4, "", true);
419
420        writer.println("\n");
421        writer.flush();
422
423        if (message != null) {
424            throw new RuntimeException(message);
425        } else {
426            throw new RuntimeException();
427        }
428    }

430    /**
431     * Static constructor using Yaml hashmap
432     *
433     * @param configVars config vars from Yaml file
434     * @return Config object with values from Yaml
435     * @throws java.text.ParseException If the Yaml does not parse correctly.
436     */
437    public static Config configFromFile(final Map<String, String> configVars) throws java.text.ParseException {
438        final Config c = new Config();
439        int lineNumber = 0;
440        for (Map.Entry<String, String> entry : configVars.entrySet()) {
441            logger.debug("config map entry is ({}) and value ({})", entry.getKey(), entry.getValue());
442            lineNumber += 1;
443            if (entry.getKey().equalsIgnoreCase("mode")) {
444                if (entry.getValue().equalsIgnoreCase("import") || entry.getValue().equalsIgnoreCase("export")) {
445                    c.setMode(entry.getValue());
446                } else {
447                    throw new java.text.ParseException(
448                        String.format("Invalid value for \"mode\": {}", entry.getValue()), lineNumber);
449                }
450            } else if (entry.getKey().equalsIgnoreCase("resource")) {
451                c.setResource(entry.getValue());
452            } else if (entry.getKey().equalsIgnoreCase("source")) {
453                c.setSource(entry.getValue());
454            } else if (entry.getKey().equalsIgnoreCase("dir")) {
455                c.setBaseDirectory(entry.getValue());
456            } else if (entry.getKey().equalsIgnoreCase("rdfLang")) {
457                c.setRdfLanguage(entry.getValue());
458            } else if (entry.getKey().trim().equalsIgnoreCase("binaries")) {
459                if (entry.getValue().equalsIgnoreCase("true") || entry.getValue().equalsIgnoreCase("false")) {
460                    c.setIncludeBinaries(Boolean.parseBoolean(entry.getValue()));
461                } else {
462                    throw new java.text.ParseException(String.format(
463                        "binaries configuration parameter only accepts \"true\" or \"false\", \"{}\" received",
464                        entry.getValue()), lineNumber);
465                }
466            } else if (entry.getKey().equalsIgnoreCase(BAG_PROFILE_OPTION_KEY)) {
467                c.setBagProfile(entry.getValue().toLowerCase());
468            } else if (entry.getKey().equalsIgnoreCase(BAG_CONFIG_OPTION_KEY)) {
469                c.setBagConfigPath(entry.getValue().toLowerCase());
470            } else if (entry.getKey().equalsIgnoreCase("predicates")) {
471                c.setPredicates(entry.getValue().split(","));
472            } else {
473                throw new java.text.ParseException(String.format("Unknown configuration key: {}", entry.getKey()),
474                    lineNumber);
475            }
476        }
477        return c;
478    }

480    /**
481     * Generate a HashMap suitable for serializing to Yaml from a Config
482     *
483     * @param config the config to turn into a Hashmap
484     * @return Map key value pairs of configuration
485     */
486    public static Map<String, String> getMap(final Config config) {
487        final Map<String, String> map = new HashMap<String, String>();
488        map.put("mode", (config.isImport() ? "import" : "export"));
489        map.put("resource", config.getResource().toString());
490        if (!config.getSource().toString().isEmpty()) {
491            map.put("source", config.getSource().toString());
492        }
493        map.put("dir", config.getBaseDirectory().getAbsolutePath());
494        if (!config.getRdfLanguage().isEmpty()) {
495            map.put("rdfLang", config.getRdfLanguage());
496        }
497        map.put("binaries", Boolean.toString(config.isIncludeBinaries()));
498        if (config.getBagProfile() != null) {
499            map.put(BAG_PROFILE_OPTION_KEY, config.getBagProfile());
500        }
501        if (config.getBagConfigPath() != null) {
502            map.put(BAG_CONFIG_OPTION_KEY, config.getBagConfigPath());
503        }
504        final String predicates = Arrays.stream(config.getPredicates()).collect(Collectors.joining(","));
505        map.put("predicates", predicates);
506        return map;
507    }
