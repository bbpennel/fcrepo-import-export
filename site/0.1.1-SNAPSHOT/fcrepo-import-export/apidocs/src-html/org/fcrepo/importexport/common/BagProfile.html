018package org.fcrepo.importexport.common;
019
020import static org.fcrepo.importexport.common.FcrepoConstants.BAG_INFO_FIELDNAME;
021import static org.slf4j.LoggerFactory.getLogger;
022
023import java.io.IOException;
024import java.io.InputStream;
025import java.util.HashMap;
026import java.util.HashSet;
027import java.util.Iterator;
028import java.util.Map;
029import java.util.Set;
030
031import org.slf4j.Logger;
032
033import com.fasterxml.jackson.databind.JsonNode;
034import com.fasterxml.jackson.databind.ObjectMapper;
035
036/**
037 * @author escowles
038 * @since 2016-12-12
039 */
040public class BagProfile {
041
042    private static final Logger logger = getLogger(BagProfile.class);
043
044    private Set<String> payloadDigestAlgorithms;
045    private Set<String> tagDigestAlgorithms;
046
047    private Set<String> sections = new HashSet<String>();
048
049    private Map<String, Map<String, Set<String>>> metadataFields = new HashMap<String, Map<String, Set<String>>>();
050
051    /**
052     * Default constructor.
053     * @param in InputStream containing the Bag profile JSON document
054     * @throws IOException when there is an I/O error reading JSON
055     */
056    public BagProfile(final InputStream in) throws IOException {
057        final ObjectMapper mapper = new ObjectMapper();
058        final JsonNode json = mapper.readTree(in);
059
060        payloadDigestAlgorithms = arrayValues(json, "Manifests-Required");
061        tagDigestAlgorithms = arrayValues(json, "Tag-Manifests-Required");
062        if (tagDigestAlgorithms == null) {
063            tagDigestAlgorithms = payloadDigestAlgorithms;
064        }
065
066        metadataFields.put(BAG_INFO_FIELDNAME, metadataFields(json, BAG_INFO_FIELDNAME));
067        sections.add(BAG_INFO_FIELDNAME);
068
069        if (json.get("Other-Info") != null) {
070            loadOtherTags(json);
071        }
072    }
073
074    private void loadOtherTags(final JsonNode json) {
075        final JsonNode arrayTags = json.get("Other-Info");
076        if (arrayTags != null && arrayTags.isArray()) {
077            arrayTags.forEach(tag -> tag.fieldNames().forEachRemaining(sections::add));
078            final Iterator<JsonNode> arrayEntries = arrayTags.elements();
079            while (arrayEntries.hasNext()) {
080                final JsonNode entries = arrayEntries.next();
081                final Iterator<String> tagNames = entries.fieldNames();
082                while (tagNames.hasNext()) {
083                    final String tagName = tagNames.next();
084                    metadataFields.put(tagName, metadataFields(entries, tagName));
085                }
086            }
087        }
088        logger.debug("tagFiles is {}", sections);
089        logger.debug("metadataFields is {}", metadataFields);
090    }
091
092    private static Set<String> arrayValues(final JsonNode json, final String key) {
093        final JsonNode values = json.get(key);
094
095        if (values == null) {
096            return null;
097        }
098
099        final Set<String> results = new HashSet<>();
100        for (int i = 0; i < values.size(); i++) {
101            results.add(values.get(i).asText());
102        }
103        return results;
104    }
105
106    /**
107     * Loads required tags and allowed values
108     *
109     * @param json json to parse
110     * @param key key in json to load tags from
111     * @return map of tags => set of allowed values
112     */
113    private static Map<String, Set<String>> metadataFields(final JsonNode json, final String key) {
114        final JsonNode fields = json.get(key);
115
116        if (fields == null) {
117            return null;
118        }
119
120        final Map<String, Set<String>> results = new HashMap<>();
121        for (final java.util.Iterator<String> it = fields.fieldNames(); it.hasNext(); ) {
122            final String name = it.next();
123            final JsonNode field = fields.get(name);
124
125            if (field.get("required") != null && field.get("required").asBoolean()) {
126                results.put(name, arrayValues(field, "values"));
127            }
128        }
129
130        return results;
131    }
132
133    /**
134     * Get the required digest algorithms for payload manifests.
135     * @return Set of digest algorithm names
136     */
137    public Set<String> getPayloadDigestAlgorithms() {
138        return payloadDigestAlgorithms;
139    }
140
141    /**
142     * Get the required digest algorithms for tag manifests.
143     * @return Set of digest algorithm names
144     */
145    public Set<String> getTagDigestAlgorithms() {
146        return tagDigestAlgorithms;
147    }
148
149    /**
150     * Get the required Bag-Info metadata fields.
151     * @return A map of field names to a Set of acceptable values (or null when the values are restricted).
152     */
153    public Map<String, Set<String>> getMetadataFields() {
154        return getMetadataFields(BAG_INFO_FIELDNAME);
155    }
156
157    /**
158     * Get the required tags for the extra tag file
159     *
160     * @param tagFile the tag file to get tags for
161     * @return map of tag = set of acceptable values, or null if tagFile doesn't exist
162     */
163    public Map<String, Set<String>> getMetadataFields(final String tagFile) {
164        return metadataFields.get(tagFile);
165    }
166
167    /**
168     * Get all the section names in this profile, which can be used with getMetadataFields().
169     *
170     * @return set of section names
171     */
172    public Set<String> getSectionNames() {
173        return sections;
174    }
175
176    /**
177     * Validate a given BagConfig against the current profile
178     *
179     * @param config the BagConfig
180     */
181    public void validateConfig(final BagConfig config) {
182        for (final String section : sections) {
183            if (config.hasTagFile(section.toLowerCase() + ".txt")) {
184                try {
185                    ProfileValidationUtil.validate(section, getMetadataFields(section),
186                        config.getFieldsForTagFile(section.toLowerCase() + ".txt"));
187                } catch (ProfileValidationException e) {
188                    throw new RuntimeException(e.getMessage(), e);
189                }
190            } else {
191                throw new RuntimeException(String.format("Error missing section %s from bag config", section));
192            }
193        }
194    }
195}
