19  package org.fcrepo.importexport.common;
20  
21  import static org.slf4j.LoggerFactory.getLogger;
22  
23  import java.util.Arrays;
24  import java.util.HashSet;
25  import java.util.Map;
26  import java.util.Set;
27  
28  import org.apache.commons.lang3.StringUtils;
29  import org.slf4j.Logger;
30  
31  /**
32   * Utility methods for validating profiles.
33   *
34   * @author dbernstein
35   * @since Dec 14, 2016
36   */
37  public class ProfileValidationUtil {
38  
39      private static final Logger logger = getLogger(ProfileValidationUtil.class);
40  
41      /*
42       * System generated Bag Info that should be ignored by validator.
43       */
44      protected static final Set<String> SYSTEM_GENERATED_FIELD_NAMES =
45              new HashSet<>(Arrays.asList("Bagging-Date",
46                                          "Bag-Size",
47                                          "Payload-Oxum"));
48  
49      private ProfileValidationUtil() {
50      }
51  
52      /**
53       * Validates the fields against the set of required fields and their constrained values.
54       *
55       * @param profileSection describes the section of the profile that is being validated.
56       * @param requiredFields the required fields and any allowable values (if constrained).
57       * @param fields The key value pairs to be validated.
58       * @throws ProfileValidationException when the fields do not pass muster. The exception message contains a
59       *         description of all validation errors found.
60       */
61      public static void validate(final String profileSection, final Map<String, Set<String>> requiredFields,
62              final Map<String, String> fields) throws ProfileValidationException {
63          if (requiredFields != null) {
64              final StringBuilder errors = new StringBuilder();
65  
66              for (String fieldName : requiredFields.keySet()) {
67                  // ignore validation on system generated fields
68                  if (SYSTEM_GENERATED_FIELD_NAMES.contains(fieldName)) {
69                      logger.debug("skipping system generated field {}...", fieldName);
70                      continue;
71                  }
72  
73                  if (fields.containsKey(fieldName)) {
74                      final String value = fields.get(fieldName);
75                      final Set<String> validValues = requiredFields.get(fieldName);
76                      if (validValues != null && !validValues.isEmpty()) {
77                          if (!validValues.contains(value)) {
78                              errors.append("\"" + value + "\" is not valid for \"" + fieldName + "\". Valid values: " +
79                                      StringUtils.join(validValues, ",") + "\n");
80                          }
81                      }
82  
83                  } else {
84                      errors.append("\"" + fieldName + "\" is a required field.\n");
85                  }
86              }
87  
88              if (errors.length() > 0) {
89                  throw new ProfileValidationException(
90                          "Bag profile validation failure: The following errors occurred in the " +
91                                  profileSection + ":\n" + errors.toString());
92              }
93          }
94  
95      }
96  
97  }
