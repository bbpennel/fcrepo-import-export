113     public Importer(final Config config, final FcrepoClient.FcrepoClientBuilder clientBuilder) {
114         this.config = config;
115         this.clientBuilder = clientBuilder;
116         this.importLogger = config.getAuditLog();
117         if (config.getBagProfile() == null) {
118             this.bag = null;
119             this.sha1 = null;
120             this.sha1FileMap = null;
121         } else {
122 
123             try {
124                 final File bagdir = config.getBaseDirectory().getParentFile();
125                 // TODO: Maybe use this once we get an updated release of bagit-java library
126                 //if (verifyBag(bagdir)) {
127                 final Path manifestPath = Paths.get(bagdir.getAbsolutePath()).resolve("manifest-SHA1.txt");
128                 this.sha1FileMap = TransferProcess.getSha1FileMap(bagdir, manifestPath);
129                 this.sha1 = MessageDigest.getInstance("SHA-1");
130                 // }
131             } catch (NoSuchAlgorithmException e) {
132                 // never happens with known algorithm names
133             }
134         }
135     }

145      * This method does the import
146      */
147     @Override
148     public void run() {
149         logger.info("Running importer...");
150         final File importContainerMetadataFile = fileForContainerURI(config.getResource());
151         importContainerDirectory = TransferProcess.directoryForContainer(config.getResource(),
152                 config.getBaseDirectory());
153 
154         discoverMembershipResources(importContainerDirectory);
155 
156         if (!importContainerMetadataFile.exists()) {
157             logger.debug("No container exists in the metadata directory {} for the requested resource {},"
158                     + " importing all contained resources instead.", importContainerMetadataFile.getPath(),
159                     config.getResource());
160         } else {
161             importFile(importContainerMetadataFile);
162         }
163         importDirectory(importContainerDirectory);
164 
165         importMembershipResources();
166         importLogger.info("Finished import... {} resources imported", successCount.get());
167     }
168 
169     private void discoverMembershipResources(final File dir) {
170         if (dir.listFiles() != null) {
171             stream(dir.listFiles()).filter(File::isFile).forEach(f -> parseMembershipResources(f));
172             stream(dir.listFiles()).filter(File::isDirectory).forEach(d -> discoverMembershipResources(d));
173         }
174     }
175 
176     private void parseMembershipResources(final File f) {
177         // skip files that aren't RDF
178         if (!f.getName().endsWith(config.getRdfExtension())) {
179             return;
180         }
181 
182         try {
183             final Model model = parseStream(new FileInputStream(f));
184             if (model.contains(null, MEMBERSHIP_RESOURCE, (RDFNode)null)) {
185                     model.listObjectsOfProperty(MEMBERSHIP_RESOURCE).forEachRemaining(node -> {
186                         logger.info("Membership resource: {}", node);
187                         membershipResources.add(URI.create(node.toString()));
188                     });
189             }
190         } catch (final IOException e) {
191             throw new RuntimeException("Error reading file: " + f.getAbsolutePath() + ": " + e.toString());
192         } catch (final RiotException e) {
193             throw new RuntimeException("Error parsing RDF: " + f.getAbsolutePath() + ": " + e.toString());
194         }
195     }
196 
197     private void importMembershipResources() {
198         membershipResources.stream().forEach(uri -> importMembershipResource(uri));
199     }
200 
201     private void importMembershipResource(final URI uri) {
202         final File f = fileForContainerURI(uri);
203         try {
204             final Model diskModel = parseStream(new FileInputStream(f));
205             final Model repoModel = parseStream(client().get(uri).perform().getBody());
206             final FcrepoResponse response = importContainer(uri, sanitize(diskModel.difference(repoModel)));
207             if (response.getStatusCode() == 401) {
208                 importLogger.error("Error importing {} to {}, 401 Unauthorized", f.getAbsolutePath(), uri);
209                 throw new AuthenticationRequiredRuntimeException();
210             } else if (response.getStatusCode() > 204 || response.getStatusCode() < 200) {
211                 importLogger.error("Error importing {} to {}, received {}", f.getAbsolutePath(), uri,
212                     response.getStatusCode());
213                 throw new RuntimeException("Error while importing membership resource " + f.getAbsolutePath()
214                         + " (" + response.getStatusCode() + "): " + IOUtils.toString(response.getBody()));
215             } else {
216                 logger.info("Imported membership resource {}: {}", f.getAbsolutePath(), uri);
217                 importLogger.info("import {} to {}", f.getAbsolutePath(), uri);
218                 successCount.incrementAndGet();
219             }
220         } catch (FcrepoOperationFailedException ex) {
221             importLogger.error(
222                 String.format("Error importing: {} to {}, Message: {}", f.getAbsolutePath(), uri, ex.getMessage()), ex);
223             throw new RuntimeException("Error importing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
224         } catch (IOException ex) {
225             importLogger.error(
226                 String.format("Error reading/parsing file: {}, Message: {}", f.getAbsolutePath(), ex.getMessage()), ex);
227             throw new RuntimeException(
228                     "Error reading or parsing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
229         }
230     }

232     private void importDirectory(final File dir) {
233         // process all the files first (because otherwise they might be
234         // created as peartree nodes which can't be updated with properties
235         // later.
236         if (dir.listFiles() != null) {
237             stream(dir.listFiles()).filter(File::isFile).forEach(file -> importFile(file));
238             stream(dir.listFiles()).filter(File::isDirectory).forEach(directory -> importDirectory(directory));
239         }
240     }

242     private void importFile(final File f) {
243         // The path, relative to the base in the export directory.
244         // This is used in place of the full path to make the output more readable.
245         final String sourceRelativePath =
246                 config.getBaseDirectory().toPath().relativize(f.toPath()).toString();
247         final String filePath = f.getPath();
248         if (filePath.endsWith(BINARY_EXTENSION) || filePath.endsWith(EXTERNAL_RESOURCE_EXTENSION)) {
249             // ... this is only expected to happen when binaries and metadata are written to the same directory...
250             if (config.isIncludeBinaries()) {
251                 logger.debug("Skipping binary {}: it will be imported when its metadata is imported.",
252                         sourceRelativePath);
253             } else {
254                 logger.debug("Skipping binary {}", sourceRelativePath);
255             }
256             return;
257         } else if (!filePath.endsWith(config.getRdfExtension())) {
258             // this could be hidden files created by the OS
259             logger.info("Skipping file with unexpected extension ({}).", sourceRelativePath);
260             return;
261         } else {
262 
263             FcrepoResponse response = null;
264             URI destinationUri = null;
265             try {
266                 final Model model = parseStream(new FileInputStream(f));
267                 if (model.contains(null, RDF_TYPE, createResource(REPOSITORY_NAMESPACE + "RepositoryRoot"))) {
268                     logger.debug("Skipping import of repository root.");
269                     return;
270                 }
271                 final ResIterator binaryResources = model.listResourcesWithProperty(RDF_TYPE, NON_RDF_SOURCE);
272                 if (binaryResources.hasNext()) {
273                     if (!config.isIncludeBinaries()) {
274                         return;
275                     }
276                     destinationUri = new URI(binaryResources.nextResource().getURI());
277                     logger.info("Importing binary {}", sourceRelativePath);
278                     response = importBinary(destinationUri, model);
279                 } else {
280                     destinationUri = new URI(config.getResource().toString() + "/"
281                             + uriPathForFile(f, importContainerDirectory));
282                     if (membershipResources.contains(destinationUri)) {
283                         logger.warn("Skipping Membership Resource: {}", destinationUri);
284                         return;
285                     }
286                     if (model.contains(null, RDF_TYPE, PAIRTREE)) {
287                         logger.info("Skipping PairTree Resource: {}", destinationUri);
288                         return;
289                     }
290 
291                     logger.info("Importing container {} to {}", f.getAbsolutePath(), destinationUri);
292                     response = importContainer(destinationUri, sanitize(model));
293 
294                 }
295 
296                 if (response == null) {
297                     logger.warn("Failed to import {}", f.getAbsolutePath());
298                 } else if (response.getStatusCode() == 401) {
299                     importLogger.error("Error importing {} to {}, 401 Unauthorized", f.getAbsolutePath(),
300                         destinationUri);
301                     throw new AuthenticationRequiredRuntimeException();
302                 } else if (response.getStatusCode() > 204 || response.getStatusCode() < 200) {
303                     final String message = "Error while importing " + f.getAbsolutePath() + " ("
304                             + response.getStatusCode() + "): " + IOUtils.toString(response.getBody());
305                     logger.error(message);
306                     importLogger.error("Error importing {} to {}, received {}", f.getAbsolutePath(), destinationUri,
307                         response.getStatusCode());
308                 } else {
309                     logger.info("Imported {}: {}", f.getAbsolutePath(), destinationUri);
310                     importLogger.info("import {} to {}", f.getAbsolutePath(), destinationUri);
311                     successCount.incrementAndGet();
312                 }
313             } catch (FcrepoOperationFailedException ex) {
314                 importLogger.error(String.format("Error importing {} to {}, Message: {}", f.getAbsolutePath(),
315                     destinationUri, ex.getMessage()), ex);
316                 throw new RuntimeException("Error importing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
317             } catch (IOException ex) {
318                 importLogger.error(String.format("Error reading/parsing {} to {}, Message: {}", f.getAbsolutePath(),
319                     destinationUri, ex.getMessage()), ex);
320                 throw new RuntimeException(
321                         "Error reading or parsing " + f.getAbsolutePath() + ": " + ex.toString(), ex);
322             } catch (URISyntaxException ex) {
323                 importLogger.error(
324                     String.format("Error building URI for {}, Message: {}", f.getAbsolutePath(), ex.getMessage()), ex);
325                 throw new RuntimeException("Error building URI for " + f.getAbsolutePath() + ": " + ex.toString(), ex);
326             }
327         }
328     }

330     private Model parseStream(final InputStream in) throws IOException {
331         final URI source = config.getSource();
332         final SubjectMappingStreamRDF mapper = new SubjectMappingStreamRDF(source, config.getResource());
333         try (final InputStream in2 = in) {
334             RDFDataMgr.parse(mapper, in2, contentTypeToLang(config.getRdfLanguage()));
335         }
336         return mapper.getModel();
337     }

339     private FcrepoResponse importBinary(final URI binaryURI, final Model model)
340             throws FcrepoOperationFailedException, IOException {
341         final Resource binaryRes = createResource(binaryURI.toString());
342         final String contentType = model.getProperty(binaryRes, HAS_MIME_TYPE).getString();
343         final boolean external = contentType.contains("message/external-body");
344         final File binaryFile =  fileForBinaryURI(binaryURI, external);
345         PutBuilder builder = client().put(binaryURI).body(binaryFile, contentType);
346         if (!external) {
347             if (sha1FileMap != null) {
348                 // Use the bagIt checksum
349                 final String checksum = sha1FileMap.get(binaryFile);
350                 logger.debug("Using Bagit checksum ({}) for file ({})", checksum, binaryFile.getPath());
351                 builder = builder.digest(checksum);
352             } else {
353                 builder = builder.digest(model.getProperty(binaryRes, HAS_MESSAGE_DIGEST)
354                                           .getObject().toString().replaceAll(".*:",""));
355             }
356         }
357         final FcrepoResponse binaryResponse = builder.perform();
358 
359         if (binaryResponse.getStatusCode() == 201 || binaryResponse.getStatusCode() == 204) {
360             logger.info("Imported binary: {}", binaryURI);
361             importLogger.info("import {} to {}", binaryFile.getAbsolutePath(), binaryURI);
362             successCount.incrementAndGet();
363 
364             final URI descriptionURI = binaryResponse.getLinkHeaders("describedby").get(0);
365             return client().put(descriptionURI).body(modelToStream(sanitize(model)), config.getRdfLanguage())
366                 .preferLenient().perform();
367         } else {
368             logger.error("Error while importing {} ({}): {}", binaryFile.getAbsolutePath(),
369                     binaryResponse.getStatusCode(), IOUtils.toString(binaryResponse.getBody()));
370             return null;
371         }
372     }

374     private FcrepoResponse importContainer(final URI uri, final Model model) throws FcrepoOperationFailedException {
375         PutBuilder builder = client().put(uri).body(modelToStream(model), config.getRdfLanguage());
376         if (sha1FileMap != null && config.getBagProfile() != null) {
377             // Use the bagIt checksum
378             final File baseDir = config.getBaseDirectory();
379             final File containerFile = Paths.get(fileForContainerURI(uri).toURI()).normalize().toFile();
380             final String checksum = sha1FileMap.get(containerFile);
381             logger.debug("Using Bagit checksum ({}) for file ({})", checksum, containerFile.getPath());
382             builder = builder.digest(checksum);
383         }
384         return builder.preferLenient().perform();
385     }

387     private Model sanitize(final Model model) throws IOException, FcrepoOperationFailedException {
388         final List<Statement> remove = new ArrayList<>();
389         for (final StmtIterator it = model.listStatements(); it.hasNext(); ) {
390             final Statement s = it.nextStatement();
391 
392             if (s.getPredicate().getNameSpace().equals(REPOSITORY_NAMESPACE)
393                     || s.getSubject().getURI().endsWith("fcr:export?format=jcr/xml")
394                     || s.getSubject().getURI().equals(REPOSITORY_NAMESPACE + "jcr/xml")
395                     || s.getPredicate().equals(DESCRIBEDBY)
396                     || s.getPredicate().equals(CONTAINS)
397                     || s.getPredicate().equals(HAS_MESSAGE_DIGEST)
398                     || s.getPredicate().equals(HAS_SIZE)
399                     || (s.getPredicate().equals(RDF_TYPE)
400                         && s.getResource().getNameSpace().equals(REPOSITORY_NAMESPACE)) ) {
401                 remove.add(s);
402             } else if (s.getObject().isResource()) {
403                 // make sure that referenced repository objects exist
404                 final String obj = s.getResource().toString();
405                 if (obj.startsWith(config.getResource().toString())) {
406                     ensureExists(URI.create(obj));
407                 }
408             }
409         }
410         return model.remove(remove);
411     }

413     /**
414      * Make sure that a URI exists in the repository.
415      */
416     private void ensureExists(final URI uri) throws IOException, FcrepoOperationFailedException {
417         try (FcrepoResponse response = client().head(uri).perform()) {
418             if (response.getStatusCode() != 200) {
419                 makePlaceholder(uri);
420             }
421         }
422     }
423 
424     private void makePlaceholder(final URI uri) throws IOException, FcrepoOperationFailedException {
425         ensureExists(parent(uri));
426 
427         final FcrepoResponse response;
428         if (fileForBinaryURI(uri, false).exists() || fileForBinaryURI(uri, true).exists()) {
429             response = client().put(uri).body(new ByteArrayInputStream(new byte[]{})).perform();
430         } else if (fileForContainerURI(uri).exists()) {
431             response = client().put(uri).body(new ByteArrayInputStream(
432                     "<> a <http://www.w3.org/ns/ldp#Container> .".getBytes()), "text/turtle").perform();
433         } else {
434             return;
435         }
436 
437         if (response.getStatusCode() != 201) {
438             logger.error("Unexpected response when creating {} ({}): {}", uri,
439                     response.getStatusCode(), response.getBody());
440         }
441     }

443     private static URI parent(final URI uri) {
444         String s = uri.toString();
445         if (s.endsWith("/")) {
446             s = s.substring(0, s.length() - 1);
447         }
448         return URI.create(s.substring(0, s.lastIndexOf("/")));
449     }

451     private InputStream modelToStream(final Model model) {
452         final ByteArrayOutputStream buf = new ByteArrayOutputStream();
453         model.write(buf, config.getRdfLanguage());
454         return new ByteArrayInputStream(buf.toByteArray());
455     }

457     private String uriPathForFile(final File f, final File baseDir) throws URISyntaxException {
458         String relative = baseDir.toPath().relativize(f.toPath()).toString();
459         relative = TransferProcess.decodePath(relative);
460 
461         // for exported RDF, just remove the ".extension" and you have the encoded path
462         if (relative.endsWith(config.getRdfExtension())) {
463             relative = relative.substring(0, relative.length() - config.getRdfExtension().length());
464         }
465 
466         return relative;
467     }

469     private File fileForBinaryURI(final URI uri, final boolean external) {
470         return new File(config.getBaseDirectory() + TransferProcess.decodePath(uri.getPath()) +
471                     (external ? EXTERNAL_RESOURCE_EXTENSION : BINARY_EXTENSION));
472     }
473 
474     private File fileForContainerURI(final URI uri) {
475         return TransferProcess.fileForURI(uri, config.getBaseDirectory(), config.getRdfExtension());
476     }

478     /**
479      * Verify the bag we are going to import
480      *
481      * @param bagDir root directory of the bag
482      * @return true if valid
483      */
484     public static boolean verifyBag(final File bagDir) {
485         try {
486             final Bag bag = BagReader.read(bagDir);
487             BagVerifier.isValid(bag, true);
488             return true;
489         } catch (Exception e) {
490             throw new RuntimeException(String.format("Error verifying bag: %s", e.getMessage()), e);
491         }
